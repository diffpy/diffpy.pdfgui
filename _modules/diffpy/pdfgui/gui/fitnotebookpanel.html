    def refresh(self):
        """Refresh the panels."""
        if not self.fit:
            return
        panel = self.fitnotebook.GetCurrentPage()
        panel.mainFrame = self.mainFrame
        panel.refresh()
        panel.fit = self.fit
        panel.parameters = self.fit.parameters
        panel.refresh()
